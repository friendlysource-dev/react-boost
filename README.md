## Install

```bash
npm install --save @fs/react-boost
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from '@fs/react-boost'
import '@fs/react-boost/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
